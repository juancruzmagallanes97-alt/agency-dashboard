**Paso 2 — Crear apps/dashboard/app/login/page.tsx:**

Esta página es Client Component ('use client') porque:
- Usa useState para loading y error
- Usa el submit handler para llamar signIn

```typescript
'use client'
import { useState } from 'react'
import { signIn } from 'next-auth/react'
import { Loader2 } from 'lucide-react'

export default function LoginPage() {
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState('')

  async function handleSubmit(e: React.FormEvent<HTMLFormElement>) {
    e.preventDefault()
    setLoading(true)
    setError('')

    const formData = new FormData(e.currentTarget)
    const email = formData.get('email') as string
    const password = formData.get('password') as string

    const result = await signIn('credentials', {
      email,
      password,
      redirect: false,
    })

    if (result?.error) {
      setError('Email o contraseña incorrectos. Verificá tus credenciales.')
      setLoading(false)
    } else {
      // Redirigir al dashboard — usar window.location para compatibilidad con next-auth v5
      window.location.href = '/'
    }
  }

  return (
    <div style={{ width: '100%', maxWidth: 360, padding: '0 24px' }}>
      {/* Card */}
      <div
        style={{
          padding: 40,
          border: '1px solid var(--border)',
          borderRadius: 12,
          boxShadow: '0 4px 16px rgba(0,0,0,0.06)',
          background: 'var(--bg)',
        }}
      >
        {/* Logo mark */}
        <div
          style={{
            width: 32,
            height: 32,
            borderRadius: 8,
            background: 'var(--accent)',
            display: 'flex',
            alignItems: 'center',
            justifyContent: 'center',
            marginBottom: 24,
            color: '#fff',
            fontSize: 14,
            fontWeight: 600,
          }}
        >
          A
        </div>

        {/* Heading */}
        <h1 style={{ fontSize: 20, fontWeight: 600, color: 'var(--text-1)', margin: 0 }}>
          Agency Intelligence
        </h1>
        <p style={{ fontSize: 14, fontWeight: 400, color: 'var(--text-2)', marginTop: 8, marginBottom: 32 }}>
          Accedé a tu panel de control
        </p>

        {/* Error state */}
        {error && (
          <div
            style={{
              background: '#FFF0F0',
              border: '1px solid #FFC5C5',
              borderRadius: 6,
              padding: '8px 16px',
              marginBottom: 16,
              fontSize: 14,
              color: 'var(--critical)',
            }}
          >
            {error}
          </div>
        )}

        {/* Form */}
        <form onSubmit={handleSubmit}>
          {/* Email field */}
          <div style={{ marginBottom: 16 }}>
            <label
              htmlFor="email"
              style={{ display: 'block', fontSize: 12, fontWeight: 400, color: 'var(--text-1)', marginBottom: 8 }}
            >
              Email
            </label>
            <input
              id="email"
              name="email"
              type="email"
              required
              autoComplete="email"
              style={{
                width: '100%',
                height: 36,
                border: '1px solid var(--border)',
                borderRadius: 6,
                padding: '0 12px',
                fontSize: 14,
                color: 'var(--text-1)',
                background: 'var(--bg)',
                outline: 'none',
                boxSizing: 'border-box',
              }}
              onFocus={e => {
                e.currentTarget.style.borderColor = 'var(--accent)'
                e.currentTarget.style.boxShadow = '0 0 0 2px rgba(35,131,226,0.2)'
              }}
              onBlur={e => {
                e.currentTarget.style.borderColor = 'var(--border)'
                e.currentTarget.style.boxShadow = 'none'
              }}
            />
          </div>

          {/* Password field */}
          <div style={{ marginBottom: 24 }}>
            <label
              htmlFor="password"
              style={{ display: 'block', fontSize: 12, fontWeight: 400, color: 'var(--text-1)', marginBottom: 8 }}
            >
              Contraseña
            </label>
            <input
              id="password"
              name="password"
              type="password"
              required
              autoComplete="current-password"
              style={{
                width: '100%',
                height: 36,
                border: '1px solid var(--border)',
                borderRadius: 6,
                padding: '0 12px',
                fontSize: 14,
                color: 'var(--text-1)',
                background: 'var(--bg)',
                outline: 'none',
                boxSizing: 'border-box',
              }}
              onFocus={e => {
                e.currentTarget.style.borderColor = 'var(--accent)'
                e.currentTarget.style.boxShadow = '0 0 0 2px rgba(35,131,226,0.2)'
              }}
              onBlur={e => {
                e.currentTarget.style.borderColor = 'var(--border)'
                e.currentTarget.style.boxShadow = 'none'
              }}
            />
          </div>

          {/* Submit button */}
          <button
            type="submit"
            disabled={loading}
            style={{
              width: '100%',
              height: 36,
              background: 'var(--accent)',
              color: '#fff',
              borderRadius: 6,
              fontSize: 14,
              fontWeight: 600,
              border: 'none',
              cursor: loading ? 'not-allowed' : 'pointer',
              opacity: loading ? 0.7 : 1,
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'center',
              gap: 8,
              transition: 'background 80ms ease, opacity 80ms ease',
            }}
          >
            {loading ? (
              <>
                <Loader2 className="animate-spin" style={{ width: 14, height: 14 }} />
                Ingresando...
              </>
            ) : (
              'Ingresar al panel'
            )}
          </button>
        </form>

        {/* Footer static link */}
        <p style={{ fontSize: 12, color: 'var(--text-2)', textAlign: 'center', marginTop: 16 }}>
          ¿Olvidaste tu contraseña?
        </p>
      </div>
    </div>
  )
}
```

**NOTA sobre signIn con redirect: false:**
Con `redirect: false`, signIn retorna un objeto `{ error, ok, url }`. Si no hay error, redirigir manualmente. Con redirect: true (default), next-auth redirige automáticamente — pero usar redirect: false da mejor control del estado loading/error.

**NOTA sobre SessionProvider:**
Según RESEARCH.md Pitfall 2 / Open Question 2: en Phase 1, el Sidebar solo llama `signOut()` que NO requiere SessionProvider. El `signIn()` en login/page.tsx tampoco requiere SessionProvider — es una llamada directa. NO agregar SessionProvider en layout.tsx.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → signIn('credentials') | Credenciales viajan como POST al Route Handler de next-auth |
| signIn → auth.ts authorize() | authorize() compara password contra bcrypt hash en env |
| auth.ts → JWT cookie | next-auth crea cookie HttpOnly Secure SameSite=lax |
| Browser refresh → proxy.ts | proxy.ts verifica JWT cookie en cada request |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Spoofing | login/page.tsx — submit form | mitigate | signIn con `redirect: false` valida en el servidor (Server Action / Route Handler de next-auth); el cliente no puede falsificar el resultado |
| T-04-02 | Information Disclosure | login/page.tsx — error message | mitigate | El mensaje de error "Email o contraseña incorrectos" NO revela si el email existe o no — mitigación de user enumeration implementada con mensaje genérico |
| T-04-03 | Tampering | auth.ts — bcryptjs.compare | mitigate | bcryptjs.compare() es time-constant — no es vulnerable a timing attacks al comparar passwords |
| T-04-04 | Elevation of Privilege | JWT cookie — acceso desde JavaScript | mitigate | next-auth v5 usa HttpOnly cookie por defecto — JavaScript del browser no puede leer el token; XSS no puede robar la sesión |
| T-04-05 | Spoofing | CSRF en login form | mitigate | next-auth v5 maneja CSRF automáticamente para CredentialsProvider — el token CSRF es incluido en el POST |
| T-04-06 | Elevation of Privilege | proxy.ts — rutas sin protección | mitigate | matcher cubre todas las rutas excepto _next/static, _next/image, favicon.ico, api/auth — no hay rutas protegidas omitidas |
</threat_model>
